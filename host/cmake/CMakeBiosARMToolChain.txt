include (CMakeForceCompiler)
set(CMAKE_SYSTEM_NAME Generic)
CMAKE_FORCE_C_COMPILER   ($ENV{GCC_ARM_NONE_TOOLCHAIN}/bin/arm-none-eabi-gcc GNU)
CMAKE_FORCE_CXX_COMPILER ($ENV{GCC_ARM_NONE_TOOLCHAIN}/bin/arm-none-eabi-g++ GNU)

# Linux build paths

IF (NOT DEFINED XDC_INSTALL_PATH)
FILE(GLOB XDC_INSTALL_PATH  ${PSDK_RTOS}/xdctools_*)

ENDIF()

IF (NOT DEFINED BIOS_INSTALL_PATH)
FILE(GLOB BIOS_INSTALL_PATH ${PSDK_RTOS}/bios_*)
SET(BIOS_INSTALL_PATH       ${BIOS_INSTALL_PATH}/packages)
ENDIF()

IF (NOT DEFINED IPC_INSTALL_PATH)
FILE(GLOB IPC_INSTALL_PATH          ${PSDK_RTOS}/ipc_*)
SET(IPC_INSTALL_PATH                ${IPC_INSTALL_PATH}/packages)
ENDIF()

SET(LLVM_PATH               ${ARM_LLVM_DIR}/include)
SET(LOKI_INCLUDE_PATH       /usr/include/loki)

SET(GCC_COVERAGE_C_FLAGS "-MMD -MP -mcpu=cortex-a15 -mtune=cortex-a15 -mfloat-abi=hard -Dxdc_target_types__=${BIOS_INSTALL_PATH}/gnu/targets/arm/std.h -Dxdc_target_name__=A15F -D_TI_RTOS -D_SYS_BIOS")
SET(GCC_COVERAGE_LINK_FLAGS "-nostartfiles -static -L${BIOS_INSTALL_PATH}/gnu/targets/arm/libs/install-native/arm-none-eabi/lib/fpu -lrdimon -lc -lm -lnosys")
SET(CMAKE_C_IMPLICIT_LINK_LIBRARIES "")

SET(CMAKE_SYSTEM_VERSION 1)

SET(CROSS_COMPILE          ON)

# specify the cross compiler
SET(CMAKE_C_FLAGS        "${GCC_COVERAGE_C_FLAGS} ${GCC_COVERAGE_LINK_FLAGS}")
SET(CMAKE_CXX_FLAGS      "${GCC_COVERAGE_C_FLAGS} ${GCC_COVERAGE_LINK_FLAGS}")
SET(CMAKE_EXE_LINK_FLAGS "${GCC_COVERAGE_LINK_FLAGS}")
SET(C_DEFINES "")
SET(CXX_DEFINES "")

# search for programs in the build host directories
SET(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM BOTH)
# for libraries and headers in the target directories
SET(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
SET(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
